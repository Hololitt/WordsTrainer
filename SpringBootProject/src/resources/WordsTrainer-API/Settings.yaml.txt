paths:
  /api/settings/words-trainer:
    get:
      tags: [Settings]
      summary: Получить настройки тренировки слов
      responses:
        '200':
          description: Текущие настройки пользователя
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/WordsTrainerSettings'
        '401':
          description: Пользователь не аутентифицирован
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorDTO'

    put:
      tags: [Settings]
      summary: Обновить настройки тренировки слов
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/WordsTrainerSettings'
      responses:
        '200':
          description: Настройки успешно обновлены
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/WordsTrainerSettings'
        '400':
          description: Некорректные данные
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorDTO'
        '401':
          description: Пользователь не аутентифицирован
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ErrorDTO'

components:
  schemas:
    WordsTrainerSettings:
      type: object
      required:
        - countLanguageCardsToRepeat
        - correctAnswersCountToFinish
        - flashCardsTrainingVariety
        - languageCardsWritingEnabled
        - translationRequestVariety
        - correctAnswersDecrementIfMistakeEnabled
      properties:
        countLanguageCardsToRepeat:
          type: integer
          minimum: 1
          example: 20
        correctAnswersCountToFinish:
          type: integer
          minimum: 1
          example: 5
        flashCardsTrainingVariety:
          type: string
          enum: [ALL, RANDOM, MOST_DIFFICULT]
          example: RANDOM
        languageCardsWritingEnabled:
          type: boolean
          example: true
        translationRequestVariety:
          type: string
          enum: [WORD, TRANSLATION, BOTH]
          example: BOTH
        correctAnswersDecrementIfMistakeEnabled:
          type: boolean
          example: false

    ErrorDTO:
      type: object
      properties:
        message:
          type: string
          example: UNAUTHORIZED
